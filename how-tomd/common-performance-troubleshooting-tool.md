# Common Performance Troubleshooting Tool

```text
#!/bin/bash

sudo uptime

sudo dmesg -T | tail -n 30

sudo vmstat 1 5 -S m

sudo mpstat -P ALL 1 5

sudo pidstat -t 1 5

sudo iostat -xmdz 1 5

sudo sar -n TCP,ETCP,DEV 1 5

sudo free -m

sudo ss -s

sudo swapon -s

sudo netstat -s

sudo lsof -iTCP -sTCP:ESTABLISHED

sudo ss -mop

sudo ss -i

sudo ps -ef f
```
